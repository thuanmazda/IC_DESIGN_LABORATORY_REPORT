\textbf{Cách hoạt động của mạch:} 
\begin{itemize}
    \item Mạch thực hiện phép toán cộng hoặc trừ 2 toán tử 8 bit A và B dựa vào tín hiệu lựa chọn Sel, khi Sel=0 thì mạch thực hiện phép cộng và khi Sel=1 thì mạch thực hiện phép trừ.
    \item Phát hiện Overflow: overflow được xác định bằng phép toán XOR giữa bit nhớ vào và bit nhớ ra của bit dấu: carry[7] XOR carry[6] như trên hình~\ref{fig:exp1_design}.
    \item Phát hiện Carry: Carry cuối cùng (tín hiệu carry[7] được lưu trữ và xuất ra ngoài qua Co).
\end{itemize}

\subsection{Mô tả mạch thiết kế bằng ngôn ngữ System Verilog HDL.}
\begin{lstlisting}[style={prettyverilog}, caption={\texttt{00\_src/fa.sv}}]
`timescale 1ns/1ns

module fa(input logic a,
	  input logic b,
	  input logic ci,
	  output logic s,
	  output logic co);
  assign s = a^b^ci;
  assign co=a&b | ci&a | ci&b;
endmodule
\end{lstlisting}

module tn1 (

\textbf{Giải thích:}
\begin{itemize}
    \item \textbf{Trường hợp Sel = 0 (Phép cộng):} 
    \begin{itemize}
        \item Không tràn: A = 0x23, B = 0x57 $\rightarrow$ S = 0x7A. Kết quả nằm trong khoảng biểu diễn, Co = 0, Ov = 0.
        \item Tràn số không dấu (Carry): A = 0xFE (254), B = 0x07 (7) $\rightarrow$ S = 0x05. Tổng vượt quá 255 (tràn 8-bit), hệ thống sinh cờ nhớ nên Co = 1. Hai số dương cộng lại không gây sai dấu nên Ov = 0.
        \item Tràn số có dấu (Overflow):    
        \begin{itemize}
            \item Âm + Âm = Dương: A = 0x8C (-116), B = 0xAB (-85) $\rightarrow$ Kết quả S = 0x37 (+55). Do cộng 2 số âm lại ra số dương (sai logic dấu) nên cờ tràn Ov = 1. Giá trị tràn 8-bit nên Co = 1.
            \item Dương + Dương = Âm: A = 0x01 (+1), B = 0x7F (+127) $\rightarrow$ Kết quả S = 0x80 (-128). Do cộng 2 số dương lại ra số âm nên cờ tràn Ov = 1. Co = 0.
        \end{itemize}
    \end{itemize}
    \item \textbf{Trường hợp Sel = 1 (Phép trừ):}
    % Điền giải thích ở đây. Ví dụ: Khi A nhỏ hơn B (A = 0x00, B = 0x57), cờ Co = 1 biểu thị số bị trừ nhỏ hơn số trừ...
    \begin{itemize}
        \item Phép trừ thông thường (A $\ge$ B): A = 0x43, B = 0x25 $\rightarrow$ S = 0x1E. Trong mạch bù 2, Co = 1 biểu thị phép trừ không cần mượn.
        \item Phép trừ có mượn (A < B): A = 0x34, B = 0xBC $\rightarrow$ S = 0x78. Kết quả sinh ra Co = 0, biểu thị phép trừ cần mượn.
        \item Tràn số có dấu (Overflow):
        \begin{itemize}
            \item Âm - Dương = Dương: A = 0x80 (-128), B = 0x01 (+1) $\rightarrow$ S = 0x7F (+127). Bị sai dấu nên Ov = 1.
            \item Dương - Âm = Âm: A = 0x7F (+127), B = 0xFF (-1) $\rightarrow$ S = 0x80 (-128). Bị sai dấu nên Ov = 1.
        \end{itemize}
    \end{itemize}
\end{itemize}

\textbf{Cách hoạt động của mạch:}
\begin{itemize}
    \item Mạch Accumulator hoạt động như một bộ tích lũy, liên tục thực hiện phép cộng (hoặc trừ) giá trị ngõ vào data với chính giá trị hiện tại của hệ thống (ngõ ra out). Mạch tái sử dụng bộ Adder/Subtractor từ Experiment 1 để thực hiện tính toán.
    \item Dựa vào sơ đồ thiết kế, ngõ vào data được nối với chân A của bộ cộng, trong khi giá trị phản hồi từ ngõ ra \textit{out} được nối ngược về chân B. Do đó, phép toán thực tế diễn ra bên trong mạch là \textbf{B +/- A} chứ không phải A +/- B. Điều này tương đương với biểu thức tích lũy chuẩn của mạch là \textbf{out = out +/- data}.
    \item Kết quả từ bộ cộng \textit{S, Co, Ov} được đưa vào các thanh ghi D-FF. Các thanh ghi này cập nhật trạng thái đồng bộ theo cạnh lên của xung \textit{clk}. Mạch sử dụng tín hiệu reset bất đồng bộ \textit{rst}. Khi \textit{rst = 1}, toàn bộ thanh ghi ngay lập tức bị xóa về 0 mà không cần chờ xung \textit{clk}.
    \item So sánh sự khác biệt giữa 2 thiết kế: Ở thiết kế trong \textit{Hình~\ref{fig:exp2_design_a}}, mạch chỉ có thanh ghi ở ngõ ra, còn ở thiết kế trong \textit{Hình~\ref{fig:exp2_design_b}}, mạch có thêm một thanh ghi \textit{Input Reg} ở ngõ vào \textit{data}. Thanh ghi này giúp đồng bộ hoá dữ liệu đầu vào với xung clock trước khi được đưa vào bộ Adder/Sub, giúp tăng tính ổn định và tránh rủi ro thời gian.
\end{itemize}

\textbf{Giải thích từng trường hợp:}

Trong đoạn mã testbench kiểm chứng, tín hiệu Sel = 0 được giữ cố định, do đó hệ thống hoạt động hoàn toàn như một bộ cộng dồn. Áp dụng logic B + A đã phân tích, phép toán tại mỗi chu kỳ là \textbf{out = out + data}

\begin{itemize}
    \item Quá trình khởi tạo: Khi rst = 1, toàn bộ ngõ ra output = 0x00, carry = 0, overflow = 0. Khi rst = 0, mạch bắt đầu tích lũy.
    \item Ở chu kì 1, khi data=0x70: \textbf{out = 0x00 + 0x70 = 0x70}. Kết quả nằm trong miền biểu diễn hợp lệ $\rightarrow$ carry = 0, overflow = 0.
    \item Ở chu kì 2, có data=0x25: Giá trị hiện tại 0x70 (Dương 112) + 0x25 (Dương 37) = 0x95. Trong hệ biểu diễn bù 2 (8-bit), 0x95 tương đương với giá trị Âm 107. Cộng hai số dương sinh ra số âm, vi phạm logic dấu $\rightarrow$ Cờ tràn overflow = 1. Do không vượt quá 255 nên carry = 0.
    \item Chu kỳ 3, data = 0x5A: kết quả trước có 0x95 (Âm) + 0x5A (Dương) = 0xEF (Âm). Phép cộng hai số trái dấu luôn an toàn $\rightarrow$ Cờ overflow = 0, carry = 0.
    \item Chu kỳ 4, data = 0x30: có kết quả trước 0xEF + 0x30 = 0x11F. Ngõ ra S bị cắt ngọn, chỉ lưu giữ 8-bit cuối là 0x1F. Phép toán vượt quá ngưỡng tối đa của số không dấu (255) $\rightarrow$ Sinh cờ nhớ carry = 1. Về mặt có dấu: Âm (-17) + Dương (48) = Dương (31), hợp logic $\rightarrow$ overflow = 0.
    \item Chu kỳ 5, data = 0x7F: 0x1F (Dương 31) + 0x7F (Dương 127) = 0x9E (Âm 98). Cộng hai số dương sinh ra số âm $\rightarrow$ Bật cờ overflow = 1, carry = 0.
    \item Chu kỳ 6, data = 0x90: 0x9E (Âm) + 0x90 (Âm) = 0x12E. Mạch giữ lại 0x2E (số Dương). Phép toán đồng thời vượt ngưỡng biểu diễn 8-bit $\rightarrow$ carry = 1, và vi phạm quy tắc cộng hai số âm sinh ra số dương $\rightarrow$ overflow = 1.
\end{itemize}

        \item Về độ trễ: Thiết kế 2(a) phản hồi nhanh hơn. Dữ liệu ngõ vào đi trực tiếp qua bộ cộng tổ hợp và cập nhật ra kết quả ngay tại sườn clock tiếp theo. Ngược lại, thiết kế 2(b) có độ trễ lớn hơn 1 chu kỳ clock, vì dữ liệu phải tốn một nhịp nạp vào Input Reg rồi mới được đưa vào tính toán.

Ở mạch 2(a), đường găng bị kéo dài từ nguồn cấp dữ liệu bên ngoài $\rightarrow$ đi xuyên qua bộ cộng $\rightarrow$ tới thanh ghi ngõ ra. \end{itemize}

\section{Experiment 3: FSM Implementation, Logic-Expression Style}
\subsection{Phương trình trạng thái và ngõ ra}
Dựa vào lưu đồ chuyển trạng thái, ta thiết lập được các phương trình Boolean cho logic Next-state và Output.

\textbf{\textit{Đối với mã One-hot (Sử dụng 9 biến trạng thái từ $Q_A$ đến $Q_I$):}}
\begin{itemize}
    \item \textbf{Phương trình trạng thái tiếp theo:}
    \begin{align*}
        D_A &= 0 \quad \text{(Chỉ đạt được thông qua tín hiệu Reset)} \\
        D_B &= \overline{w} \cdot (Q_A + Q_F + Q_G + Q_H + Q_I) \\
        D_C &= \overline{w} \cdot Q_B \\
        D_D &= \overline{w} \cdot Q_C \\
        D_E &= \overline{w} \cdot (Q_D + Q_E) \quad \text{(Self-loop khi } w=0 \text{)} \\
        D_F &= w \cdot (Q_A + Q_B + Q_C + Q_D + Q_E) \\
        D_G &= w \cdot Q_F \\
        D_H &= w \cdot Q_G \\
        D_I &= w \cdot (Q_H + Q_I) \quad \text{(Self-loop khi } w=1 \text{)}
    \end{align*}
    
    \item \textbf{Phương trình ngõ ra:} $z = Q_E + Q_I$
\end{itemize}

\textit{\textbf{Đối với mã Binary:}} Do sử dụng 4 biến trạng thái ($Y_3 Y_2 Y_1 Y_0$), việc rút gọn bằng bìa Karnaugh sẽ tạo ra các phương trình Boolean rất phức tạp. Do đó, phương trình next-state cho mã Binary sẽ được biểu diễn trực tiếp dưới dạng biểu thức điều kiện (Logic-expression/MUX logic) trong phần mã nguồn SystemVerilog. Phương trình ngõ ra: $z = \overline{Y_3} Y_2 \overline{Y_1} \overline{Y_0} + Y_3 \overline{Y_2} \overline{Y_1} \overline{Y_0}$.
\subsection{Mô tả mạch thiết kế bằng ngôn ngữ System Verilog HDL.}
\begin{lstlisting}[style={prettyverilog}, caption={\texttt{00\_src/fsm\_binary.sv}}]
`timescale 1ns/1ns
module fsm_binary (
    input  logic clk,
    input  logic rst,
    input  logic w,
    output logic z
);

    logic [3:0] state, next_state;

    localparam logic [3:0] A = 4'b0000, B = 4'b0001,C = 4'b0010, D = 4'b0011, E = 4'b0100, 
    F = 4'b0101, G = 4'b0110, H = 4'b0111, I = 4'b1000;

    always_ff @(posedge clk) begin
        if (rst)
            state <= A;
        else
            state <= next_state;
    end

    assign next_state =
        (state == A) ? (w ? F : B) :
        (state == B) ? (w ? F : C) :
        (state == C) ? (w ? F : D) :
        (state == D) ? (w ? F : E) :
        (state == E) ? (w ? F : E) :
        (state == F) ? (w ? G : B) :
        (state == G) ? (w ? H : B) :
        (state == H) ? (w ? I : B) :
        (state == I) ? (w ? I : B) :
        A;

    assign z = (state == E) || (state == I);

endmodule
\end{lstlisting}
\textbf{Cách hoạt động:}
Trong thiết kế này, hệ thống sử dụng thanh ghi 4-bit để lưu trữ 9 trạng thái (từ A đến I). Mạch được chia làm 3 khối logic rõ ràng:
\begin{itemize}
    \item Khối lưu trữ trạng thái: Sử dụng khối \texttt{always\_ff} để cập nhật trạng thái hiện tại (\texttt{state}) bằng trạng thái tiếp theo (\texttt{next\_state}) tại mỗi sườn lên của xung \texttt{clk} . Khi tín hiệu \texttt{rst} được kích hoạt, hệ thống lập tức trở về trạng thái khởi tạo A .
    \item Khối logic chuyển trạng thái: Do yêu cầu sử dụng phong cách \texttt{logic-expression}, mạch không sử dụng khối \texttt{always\_comb} hay lệnh \texttt{if/case}. Thay vào đó, toàn bộ logic chuyển trạng thái được mô tả bằng một chuỗi toán tử điều kiện 3 ngôi (\texttt{?:}) nối tiếp nhau trong một lệnh \texttt{assign} duy nhất. Cấu trúc này khi tổng hợp sẽ tạo ra một mạng lưới các bộ dồn kênh xếp tầng khổng lồ ở mức cổng logic.
    \item Khối logic ngõ ra: Ngõ ra $z$ là một mạch tổ hợp đơn giản, kiểm tra xem \texttt{state} hiện tại có đang bằng \texttt{E} (phát hiện 4 bit 0) hoặc bằng \texttt{I} (phát hiện 4 bit 1) hay không để xuất mức logic 1 .
\end{itemize}
\begin{lstlisting}[style={prettyverilog}, caption={\texttt{00\_src/fsm\_onehot.sv}}]
`timescale 1ns/1ns
module fsm_onehot (
    input  logic clk,
    input  logic rst,
    input  logic w,
    output logic z
);
    // state[0]=A, [1]=B, [2]=C, [3]=D, [4]=E [5]=F, [6]=G, [7]=H, [8]=I
    logic [8:0] state, next_state;
    always_ff @(posedge clk) begin
        if (rst)
            state <= 9'b000000001;
        else
            state <= next_state;
    end
    assign next_state[0] = 1'b0;
    assign next_state[1] = ~w & (state[0] | state[5] | state[6] | state[7] | state[8]);
    assign next_state[2] = ~w & state[1];
    assign next_state[3] = ~w & state[2];
    assign next_state[4] = ~w & (state[3] | state[4]);   // overlapping
    assign next_state[5] =  w & (state[0] | state[1] | state[2] | state[3] | state[4]);
    assign next_state[6] =  w & state[5];
    assign next_state[7] =  w & state[6];
    assign next_state[8] =  w & (state[7] | state[8]);   // overlapping
    // Output: z=1 khi state E (bit 4) hoac state I (bit 8)
    assign z = state[4] | state[8];

endmodule
\end{lstlisting}
\textbf{Cách hoạt động:} 
Khác với Binary, mã hóa One-hot sử dụng đến 9 bit (tương ứng với 9 Flip-flop) để biểu diễn 9 trạng thái . Tại bất kỳ thời điểm nào, chỉ có duy nhất 1 bit mang giá trị 1.
\begin{itemize}
    \item Khối lưu trữ trạng thái: Tương tự như Binary, khối \texttt{always\_ff} làm nhiệm vụ chốt trạng thái theo xung \texttt{clk} . Tuy nhiên, khi có tín hiệu \texttt{rst}, trạng thái A được khởi tạo bằng chuỗi bit \texttt{9'b000000001} (chỉ bit số 0 bật lên 1) .
    \item Khối logic chuyển trạng thái: Đây là điểm sáng của thiết kế One-hot. Mỗi bit của \texttt{next\_state} (từ bit 0 đến bit 8) được tính toán riêng biệt thông qua một lệnh \texttt{assign} độc lập . Các lệnh này tương ứng trực tiếp với các phương trình Boolean đã được rút gọn (Ví dụ: bit \texttt{next\_state[2]} chỉ bằng 1 khi ngõ vào $w = 0$ và hệ thống đang ở \texttt{state[1]}) . Cấu trúc này giúp mạch tổng hợp ra các cổng AND/OR rất nhỏ gọn, chạy ở tần số rất cao vì không bị vướng bộ MUX phức tạp như mã Binary.
    \item Khối logic ngõ ra: Rất tối ưu, hệ thống không cần dùng đến các bộ so sánh. Mạch chỉ cần dùng một cổng OR duy nhất để lấy tổng của bit số 4 (\texttt{state[4]} tương ứng với E) và bit số 8 (\texttt{state[8]} tương ứng với I).
\end{itemize}

\textbf{Giải thích:}

Dựa vào giản đồ thời gian mô phỏng, ta có thể thấy cả hai bộ mã hóa Binary và One-hot đều hoạt động song song và cho ra đáp ứng ngõ ra $z$ hoàn toàn trùng khớp nhau dưới cùng một chuỗi tín hiệu kích thích $w$. Cụ thể qua các giai đoạn:

\begin{itemize}
    \item \textbf{Khởi tạo:} Khi tín hiệu $rst = 1$, FSM bị cưỡng bức về trạng thái ban đầu A (0000 đối với Binary và 000000001 đối với One-hot). Ngõ ra $z = 0$.
    
    \item \textbf{Phát hiện chuỗi 4 bit 0:} Khi $w = 0$ liên tục trong 4 chu kỳ xung nhịp, FSM lần lượt dịch chuyển qua các trạng thái $A \rightarrow B \rightarrow C \rightarrow D \rightarrow E$. Ngay tại trạng thái E, ngõ ra $z$ bật lên 1, báo hiệu đã nhận diện thành công chuỗi 4 bit 0.
    
    \item \textbf{Tính năng chồng chéo:} Ở chu kỳ tiếp theo, nếu $w$ vẫn tiếp tục bằng 0, FSM tự lặp lại tại trạng thái E ($E \rightarrow E$) và ngõ ra $z$ vẫn duy trì mức 1. 
    
    \item \textbf{Phát hiện chuỗi 4 bit 1:} Khi $w$ chuyển sang 1, chuỗi đếm 0 bị ngắt, mạch nhảy sang trạng thái F. Tiếp tục giữ $w = 1$ trong 4 chu kỳ, mạch đi theo nhánh $F \rightarrow G \rightarrow H \rightarrow I$. Tại trạng thái I, ngõ ra $z = 1$. Tính năng chồng chéo tiếp tục hoạt động đúng nếu $w$ giữ mức 1 ở các nhịp sau đó.
    
    \item \textbf{Chuỗi bị ngắt quãng:} Khi đang đếm dở dang (ví dụ đang ở C, tức đếm được hai bit 0), nếu tín hiệu $w$ đột ngột đảo mức thành 1, hệ thống lập tức hủy chuỗi đang đếm và chuyển hướng sang nhánh đếm mới (nhảy sang F). Ngõ ra $z$ giữ nguyên mức 0, chứng minh mạch không bị nhận diện sai.
\end{itemize}

\textbf{Kết luận:} Giản đồ xung chứng minh mạch hoạt động chính xác theo đặc tả thiết kế.

\section{Experiment 4: FSM Implementation, Behavioral Style}

Ở phần này, FSM nhận diện chuỗi được thiết kế lại bằng phong cách mô tả hành vi (Behavioral Style) của SystemVerilog. Thay vì tự tính toán các phương trình logic, ta sử dụng khối \texttt{always\_comb} kết hợp với lệnh \texttt{case} để mô tả trực tiếp luồng chuyển trạng thái.

\subsection{Mô tả mạch thiết kế bằng ngôn ngữ System Verilog HDL.}

Cả hai phương pháp mã hóa One-hot và Binary đều cho cùng một hành vi đáp ứng ngõ ra $z$ theo ngõ vào $w$.

\textbf{1. Mã hóa Binary (Behavioral Style):}

% CHÚ THÍCH: Dán code SystemVerilog mã Binary vào đây (sử dụng lstlisting giống exp3)
\begin{lstlisting}[style={prettyverilog}, caption={00\_src/fsm\_binary\_behav.sv}]
`timescale 1ns/1ns
module fsm_binary_behav (
    input logic clk,
    input logic rst,
    input logic w,
    output logic z
);
    typedef enum logic [3:0] {
        A = 4'd0,
        B = 4'd1,
        C = 4'd2,
        D = 4'd3,
        E = 4'd4,
        F = 4'd5,
        G = 4'd6,
        H = 4'd7,
        I = 4'd8
    } state_t;
    
    state_t present_state, next_state;
    always_ff @(posedge clk, posedge rst) begin
        if (rst)
            present_state <= A;
        else
            present_state <= next_state;
    end

    always_comb begin
        next_state = present_state;
        case (present_state)
            A: if (w) next_state = F; else next_state = B;
            B: if (w) next_state = F; else next_state = C;
            C: if (w) next_state = F; else next_state = D;
            D: if (w) next_state = F; else next_state = E;
            E: if (w) next_state = F; else next_state = E;
            F: if (w) next_state = G; else next_state = B;
            G: if (w) next_state = H; else next_state = B;
            H: if (w) next_state = I; else next_state = B;
            I: if (w) next_state = I; else next_state = B;
            default: next_state = A;
        endcase
    end

    assign z = (present_state == E) || (present_state == I);
endmodule
\end{lstlisting}

\subsection{Hoàn thành giản đồ xung dựa theo thiết kế trên.}
Về mặt nhận diện chuỗi, mạch ở Experiment 4 thực hiện chức năng hoàn toàn giống với Experiment 3, do đó đáp ứng của ngõ ra $z$ đối với dữ liệu $w$ là trùng khớp. Tuy nhiên, có một sự khác biệt về bản chất phần cứng được thể hiện trên giản đồ thời gian: \textbf{Hành vi của tín hiệu Reset}.

\begin{itemize}
    \item Ở \textbf{Experiment 3}, hệ thống sử dụng \textbf{Reset Đồng Bộ} (\texttt{always\_ff @(posedge clk)}). Nếu tín hiệu $rst$ được kích hoạt giữa chu kỳ, hệ thống bắt buộc phải chờ đến sườn lên của xung $clk$ tiếp theo mới đưa trạng thái về $A$.
    \item Ở \textbf{Experiment 4}, thiết kế sử dụng \textbf{Reset Bất Đồng Bộ} (\texttt{always\_ff @(posedge clk, posedge rst)}). Khi quan sát kỹ giản đồ xung, ta sẽ thấy ngay khi tín hiệu $rst$ chuyển lên mức 1, toàn bộ trạng thái (\texttt{present\_state}) lập tức bị ép về trạng thái khởi tạo $A$ mà không cần quan tâm đến sườn xung $clk$.
\end{itemize}
